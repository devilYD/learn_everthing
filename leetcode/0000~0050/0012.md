class Solution {
public:
/*
I             1         0
IV            4         1
V             5         2
IX            9         3
X             10        4
XL            40        5
L             50        6
XC            90        7
C             100       8
CD            400       9
D             500       10
CM            900       11
M             1000      12
*/
    string intToRoman(int num) {
        int a[13]{};
        int b[13] = {1,4,5,9,10,40,50,90,100,400,500,90,1000};
        string c[13]={"I","IV","V","IX","X","XL","L","XC","C","CD","D","CM","M"};
        string s{};
        int n(0);

        for(int i=12;i>=0;i--)
        {
            while(num-b[i]>0)
            {
                n++;
                num=num-b[i];
            }
            a[i]=n;
            n=0;
        }

        for(int i=12;i>=0;i--)
            for(int f=0;f<a[i];f++)
                s+=c[i];

        return s;
    }
};
